1. Objetivo
Substituir o banco de dados temporário em memória (arrays JavaScript via padrão Singleton) por um banco de dados externo real hospedado no Supabase (PostgreSQL). A migração manteve a arquitetura em camadas existente (Controller → Service → Repository) e adicionou persistência permanente dos dados, além de tratamento de erros robusto em todos os endpoints.

5. Antes e Depois — Principais Mudanças

database/singleton.js

Antes — Banco em Memória
class Database {
  constructor() {
    if (!Database.instance) {
      this.usuarios = [];
      this.tarefas = [];
      Database.instance = this;
    }
    return Database.instance;
  }
}
const instance = new Database();
Object.freeze(instance);
module.exports = instance;
Depois — Cliente Supabase
const { createClient } =
  require('@supabase/supabase-js');

const supabase = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_KEY
);

module.exports = supabase;

repositories/usuarioRepository.js — método salvar()

Antes — Array em Memória
salvar(usuario) {
  db.usuarios.push(usuario);
  return usuario;
}
Depois — Supabase
async salvar(usuario) {
  const { data, error } = await supabase
    .from('usuarios')
    .insert(usuario)
    .select()
    .single();
  if (error) throw error;
  return data;
}

services/usuarioService.js — método criar()

Antes — ID Manual
let id = 1;

exports.criar = (dados) => {
  const usuario = criarUsuario(
    id++,
    dados.nome,
    dados.email
  );
  return usuarioRepository
    .salvar(usuario);
};
Depois — ID pelo PostgreSQL
exports.criar = async (dados) => {
  return usuarioRepository.salvar({
    nome: dados.nome,
    email: dados.email
  });
};

controllers/tarefaController.js — método criar()

Antes — Sem tratamento de erro
exports.criar = (req, res) => {
  const tarefa =
    tarefaService.criar(req.body);
  res.json(tarefa);
};
Depois — Async com try/catch
exports.criar = async (req, res) => {
  try {
    const tarefa = await
      tarefaService.criar(req.body);
    res.status(201).json(tarefa);
  } catch (error) {
    res.status(500).json({
      erro: error.message
    });
  }
};
